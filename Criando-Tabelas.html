Script T-SQL: Criação de Tabelas

Este script cria as tabelas tblMarcas e tblModelos no banco de dados Concessionaria.

USE Concessionaria;

-- Tabela de Marcas
CREATE TABLE tblMarcas
(
   idMarca     INT         IDENTITY       PRIMARY KEY,
   nomeMarca   NCHAR(10)   NOT NULL       UNIQUE
);

-- Visualizando estrutura da tabela de Marcas
EXEC sp_help tblMarcas;

-- Tabela de Modelos
CREATE TABLE tblModelos
(
   idModelo    INT          IDENTITY,
   idMarca     INT          NOT NULL,
   nomeModelo  NCHAR(30)    NOT NULL,

   CONSTRAINT PK_tblModelos_idModelo PRIMARY KEY (idModelo),
   CONSTRAINT FK_tblModelos_tblMarcas_idMarca FOREIGN KEY (idMarca)
       REFERENCES tblMarcas(idMarca),
   CONSTRAINT UQ_tblModelos_nomeModelo UNIQUE (nomeModelo)
);
